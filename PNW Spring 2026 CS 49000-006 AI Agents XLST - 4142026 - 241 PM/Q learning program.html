import numpy as np
import random

# Define environment
n_states = 6
n_actions = 2   # 0 = left, 1 = right

# Reward matrix
R = np.array([
    [-1, 0],
    [-1, 0],
    [-1, 0],
    [-1, 0],
    [-1, 10],  # goal reward
    [0, 0]
])

# Q-table initialization
Q = np.zeros((n_states, n_actions))

# Hyperparameters
alpha = 0.1      # learning rate
gamma = 0.9      # discount factor
epsilon = 0.2    # exploration rate
episodes = 500

# Training
for episode in range(episodes):
    state = random.randint(0, n_states - 1)

    for step in range(100):
        # Choose action (epsilon-greedy)
        # Generate random number between 0 and 1.If less than epsilon → explore.
        if random.uniform(0, 1) < epsilon:
            action = random.randint(0, n_actions - 1)
        else:
            action = np.argmax(Q[state]) #Otherwise exploit

        # Take action → next state
        next_state = min(state + 1, n_states - 1) if action == 1 else max(state - 1, 0) # which action to take to go the next state

        reward = R[state, action]

        # Q-learning update
        Q[state, action] = Q[state, action] + alpha * (
            reward + gamma * np.max(Q[next_state]) - Q[state, action]
        )

        state = next_state

# Learned Q-table
print("Q-table:")
print(Q)

# Policy extraction
policy = np.argmax(Q, axis=1)
print("Optimal policy (0=left, 1=right):")
print(policy)
