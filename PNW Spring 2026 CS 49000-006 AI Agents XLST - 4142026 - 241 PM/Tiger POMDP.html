Tiger POMDP solved with belief-space value iteration
# Runs in Google Colab

import numpy as np
import matplotlib.pyplot as plt

# -----------------------------
# POMDP PARAMETERS
# -----------------------------

states = ["TL", "TR"]
actions = ["listen", "open_left", "open_right"]
observations = ["hear_left", "hear_right"]

gamma = 0.95

# Transition model T[s, a, s']
T = {
    "listen": np.array([[1, 0],
                        [0, 1]]),

    "open_left": np.array([[0.5, 0.5],
                           [0.5, 0.5]]),   # reset

    "open_right": np.array([[0.5, 0.5],
                            [0.5, 0.5]])    # reset
}

# Observation model O[a][s'][o]
# Only informative when listening
O = {
    "listen": {
        "TL": {"hear_left": 0.85, "hear_right": 0.15},
        "TR": {"hear_left": 0.15, "hear_right": 0.85}
    },
    "open_left": {
        "TL": {"hear_left": 0.5, "hear_right": 0.5},
        "TR": {"hear_left": 0.5, "hear_right": 0.5}
    },
    "open_right": {
        "TL": {"hear_left": 0.5, "hear_right": 0.5},
        "TR": {"hear_left": 0.5, "hear_right": 0.5}
    }
}

# Reward function R[s,a]
R = {
    "listen": np.array([-1, -1]),
    "open_left": np.array([-100, +10]),
    "open_right": np.array([+10, -100])
}


# -----------------------------
# BELIEF UPDATE FUNCTION --> main part of POMDP

# -----------------------------
def belief_update(belief, action, observation):
    new_belief = np.zeros(2)

    for s_prime in range(2):
        obs_prob = O[action][states[s_prime]][observation]
        transition_sum = 0

        for s in range(2):
            transition_sum += belief[s] * T[action][s, s_prime]

        new_belief[s_prime] = obs_prob * transition_sum # belief update

    # normalize
    new_belief /= np.sum(new_belief)
    return new_belief


# -----------------------------
# VALUE ITERATION IN BELIEF SPACE
# -----------------------------

# discretize belief space
belief_points = np.linspace(0.01, 0.99, 200)
V = np.zeros(len(belief_points))
policy = ["listen"] * len(belief_points)

theta = 1e-4
max_iters = 200

for iteration in range(max_iters):
    delta = 0
    new_V = np.copy(V)

    for i, b in enumerate(belief_points):
        belief = np.array([b, 1-b])

        action_values = []

        for a in actions:
            # expected immediate reward
            r = np.dot(belief, R[a])

            # expected future value
            future_value = 0

            for obs in observations:
                updated_belief = belief_update(belief, a, obs)
                b_idx = np.argmin(np.abs(belief_points - updated_belief[0]))

                obs_prob = 0
                for s_prime in range(2):
                    obs_prob += O[a][states[s_prime]][obs] * \
                                np.dot(belief, T[a][:, s_prime])

                future_value += obs_prob * V[b_idx]

            action_values.append(r + gamma * future_value)

        best_value = max(action_values)
        best_action = actions[np.argmax(action_values)]

        new_V[i] = best_value
        policy[i] = best_action

        delta = max(delta, abs(new_V[i] - V[i]))

    V = new_V

    if delta < theta:
        print(f"Converged in {iteration} iterations")
        break


# -----------------------------
# DISPLAY OPTIMAL POLICY
# -----------------------------
print("\nOptimal policy over belief space:")
for i in range(0, len(belief_points), 20):
    print(f"Belief tiger-left={belief_points[i]:.2f} -> {policy[i]}")


# -----------------------------
# PLOT POLICY
# -----------------------------
policy_numeric = [actions.index(a) for a in policy]

plt.figure(figsize=(10,5))
plt.plot(belief_points, policy_numeric)
plt.yticks([0,1,2], actions)
plt.xlabel("Belief P(Tiger Left)")
plt.ylabel("Optimal Action")
plt.title("Optimal Policy for Tiger POMDP")
plt.grid()
plt.show()
